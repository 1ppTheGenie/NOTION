CREATE TABLE AgentProspect (

  AgentName VARCHAR(100),
  AgentEmail VARCHAR(255),
  AgentPhone VARCHAR(20),
  Brokerage VARCHAR(255),

  Rank INT, -- 1-10 (top agents in zip)
  AvgHomePrice DECIMAL(18,2),
